MATCH (follower:Artist)-[:FOLLOWS]->(targetArtist:Artist)-[:FOLLOWS]->(following:Artist)
RETURN targetArtist AS Artist, COUNT(distinct follower) AS Followers, COUNT(distinct following) AS Following


MATCH (p:Artist)-[f:FOLLOWS]->(p1:Artist)
RETURN p.id AS Artist, COLLECT(p1.id) AS Following
